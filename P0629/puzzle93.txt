minimize

general{
#それぞれが持つ桃の個数を表す変数
    miya; miwa; mina;
}

member = [miya,miwa,mina];

#「美和と美奈の桃の数は違う」という美弥の発言
comment = miwa!=mina;

subject{
#メンバー全員が桃を1個か2個持っている
    forall x in member:x==1||x==2;

#桃を1個持っている者がいる
    exists x in member:x==1;
    
#桃を2個持っている者がいる
    exists x in member:x==2;

#桃が1個の者はウソつき
    miya==1 -> !comment;

    @object==0;
}
